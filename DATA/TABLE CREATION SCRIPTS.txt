
CREATE TABLE appointments(ID INT PRIMARY KEY, TITLE VARCHAR(255), DESCRIPTION VARCHAR(255), LOCATION VARCHAR(255), TYPE VARCHAR(255), START_DATE DATE, START_TIME TIME, END_DATE DATE, END_TIME TIME, CREATED_DATE DATE, CREATED_TIME TIME, CREATED_BY VARCHAR(255), LAST_UPDATED TIMESTAMP, LAST_UPDATED_BY VARCHAR(255));

CREATE TABLE contacts(ID INT PRIMARY KEY, CONTACT_NAME VARCHAR(255), EMAIL VARCHAR(255));

ALTER TABLE CONTACTS DROP COLUMN APPOINTMENTS
ALTER TABLE CONTACTS ADD APPOINTMENTS INT 

CREATE TABLE countries(ID INT PRIMARY KEY, COUNTRY VARCHAR(255),  CREATED_DATE DATE, CREATED_TIME TIME, CREATED_BY VARCHAR(255), LAST_UPDATED TIMESTAMP, LAST_UPDATED_BY VARCHAR(255));
CREATE TABLE customers(ID INT PRIMARY KEY, CUSTOMER_NAME VARCHAR(255), ADDRESS VARCHAR(255), ZIP_CODE VARCHAR(255), PHONE_NUMBER VARCHAR(255), CREATED_DATE DATE, CREATED_TIME TIME, CREATED_BY VARCHAR(255), LAST_UPDATED TIMESTAMP, LAST_UPDATED_BY VARCHAR(255), APPOINTMENT INT);
ALTER TABLE CUSTOMERS ADD FOREIGN KEY(APPOINTMENT) REFERENCES APPOINTMENTS(ID)

CREATE TABLE first_level_divisions(ID INT PRIMARY KEY, DIVISION VARCHAR(255), CREATED_DATE DATE, CREATED_TIME TIME, CREATED_BY VARCHAR(255), LAST_UPDATED TIMESTAMP, LAST_UPDATED_BY VARCHAR(255), COUNTRY INT);
ALTER TABLE FIRST_LEVEL_DIVISIONS ADD FOREIGN KEY(COUNTRY) REFERENCES COUNTRIES(ID)


CREATE TABLE users(ID INT PRIMARY KEY, USER_NAME VARCHAR(255), PASSWORD VARCHAR(255), CREATED_DATE DATE, CREATED_TIME TIME, CREATED_BY VARCHAR(255), LAST_UPDATED TIMESTAMP, LAST_UPDATED_BY VARCHAR(255), APPOINTMENT INT);
ALTER TABLE USERS ADD FOREIGN KEY(APPOINTMENT) REFERENCES APPOINTMENTS(ID)
